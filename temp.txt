package com.abnamro.crms.lpdrisksignaladministration.config;

import static org.junit.jupiter.api.Assertions.assertEquals;
import static org.mockito.Mockito.when;

import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;
import org.junit.jupiter.api.extension.ExtendWith;
import org.mockito.InjectMocks;
import org.mockito.Mock;
import org.mockito.junit.jupiter.MockitoExtension;
import org.springframework.core.env.Environment;

@ExtendWith(MockitoExtension.class)
public class ServiceBusPropertiesTest {

    @Mock
    private Environment env;

    @InjectMocks
    private ServiceBusProperties serviceBusProperties;

    @BeforeEach
    void setUp() {
        when(env.getProperty("spring.cloud.azure.credential.client-id")).thenReturn("test-client-id");
        when(env.getProperty("spring.cloud.azure.servicebus.namespace")).thenReturn("test-namespace");
        when(env.getProperty("event.writer.queue")).thenReturn("test-queue");
    }

    @Test
    void testProperties() {
        assertEquals("test-client-id", serviceBusProperties.getCredentialClientId());
        assertEquals("test-namespace", serviceBusProperties.getServiceBusNamespace());
        assertEquals("test-queue", serviceBusProperties.getRsdQueue());
    }
}

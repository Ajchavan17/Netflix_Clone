package com.abnamro.crms.lpdrisksignaladministration.model;

import static org.junit.jupiter.api.Assertions.*;
import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;
import org.mockito.InjectMocks;
import org.mockito.junit.jupiter.MockitoExtension;
import org.junit.jupiter.api.extension.ExtendWith;

@ExtendWith(MockitoExtension.class)
class LPDControlTest {

    @InjectMocks
    private LPDControl lpdControl;

    private static final Integer LPD_CONTROL_IDENTIFIER = 123;
    private static final String FACILITY_IDENTIFIER = "FAC123";
    private static final String INPUT_MESSAGE = "Test Input Message";
    private static final String ERROR_MESSAGE = "Test Error Message";

    @BeforeEach
    void setUp() {
        lpdControl = new LPDControl();
        lpdControl.setLpdControlIdentifier(LPD_CONTROL_IDENTIFIER);
        lpdControl.setFacilityIdentifier(FACILITY_IDENTIFIER);
        lpdControl.setInputMessage(INPUT_MESSAGE);
        lpdControl.setErrorMessage(ERROR_MESSAGE);
    }

    @Test
    void testGettersAndSetters() {
        assertEquals(LPD_CONTROL_IDENTIFIER, lpdControl.getLpdControlIdentifier());
        assertEquals(FACILITY_IDENTIFIER, lpdControl.getFacilityIdentifier());
        assertEquals(INPUT_MESSAGE, lpdControl.getInputMessage());
        assertEquals(ERROR_MESSAGE, lpdControl.getErrorMessage());
    }

    @Test
    void testEqualsAndHashCode() {
        LPDControl anotherLpdControl = new LPDControl();
        anotherLpdControl.setLpdControlIdentifier(LPD_CONTROL_IDENTIFIER);
        anotherLpdControl.setFacilityIdentifier(FACILITY_IDENTIFIER);
        anotherLpdControl.setInputMessage(INPUT_MESSAGE);
        anotherLpdControl.setErrorMessage(ERROR_MESSAGE);

        assertEquals(lpdControl, anotherLpdControl);
        assertEquals(lpdControl.hashCode(), anotherLpdControl.hashCode());
    }

    @Test
    void testToString() {
        String expectedString = "LPDControl(lpdControlIdentifier=123, facilityIdentifier=FAC123, inputMessage=Test Input Message, errorMessage=Test Error Message)";
        assertEquals(expectedString, lpdControl.toString());
    }
}

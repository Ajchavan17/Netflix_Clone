import static org.junit.jupiter.api.Assertions.*;
import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;
import org.mockito.InjectMocks;
import org.mockito.MockitoAnnotations;

class PartyTest {

    @InjectMocks
    private Party party;

    @BeforeEach
    void setUp() {
        MockitoAnnotations.openMocks(this);
        party = new Party();
    }

    @Test
    void testGettersAndSetters() {
        party.setIdentifier(1);
        party.setPartyId("P001");
        party.setPartyRoleCode("ROLE123");

        assertEquals(1, party.getIdentifier());
        assertEquals("P001", party.getPartyId());
        assertEquals("ROLE123", party.getPartyRoleCode());
    }

    @Test
    void testEqualsAndHashCode() {
        Party party1 = new Party();
        Party party2 = new Party();

        party1.setIdentifier(1);
        party2.setIdentifier(1);

        assertEquals(party1, party2);
        assertEquals(party1.hashCode(), party2.hashCode());
    }

    @Test
    void testToString() {
        party.setIdentifier(1);
        party.setPartyId("P001");
        party.setPartyRoleCode("ROLE123");

        String expected = "Party(identifier=1, partyId=P001, partyRoleCode=ROLE123)";
        assertTrue(party.toString().contains("identifier=1"));
        assertTrue(party.toString().contains("partyId=P001"));
        assertTrue(party.toString().contains("partyRoleCode=ROLE123"));
    }
}
